from bs4 import BeautifulSoup

html = """
<html>
<body>
<div class="content">
    <p>This is a paragraph.</p>
    <p>This is another paragraph.</p>
</div>
</body>
</html>
"""

soup = BeautifulSoup(html, 'html.parser')

# Find all paragraphs and extract text
paragraphs = soup.find_all('p')
for p in paragraphs:
    print(p.get_text())
